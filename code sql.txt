CREATE DATABASE User
use User
-- Tạo bảng users để lưu trữ email và mật khẩu của người dùng
CREATE TABLE users (
    id INT PRIMARY KEY IDENTITY(1,1),  -- ID tự động tăng
    email NVARCHAR(255) NOT NULL UNIQUE,  -- Email là duy nhất
    password NVARCHAR(255) NOT NULL  -- Mật khẩu
);
